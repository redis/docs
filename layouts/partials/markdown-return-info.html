{{- /* Process return-info shortcodes in raw markdown content (called from `process-markdown-content.html` for the .md and .json output paths). Replaces every `{{< return-info >}}` tag with the markdown body rendered directly from the page's entry in `data/commands_returns_.json`. Single source of truth — no intermediate file reads. Input dict: RawContent: markdown body string Page: current Hugo page */ -}} {{- $content := .RawContent -}} {{- $page := .Page -}} {{- $entry := partial "returns-data.html" $page -}} {{- if $entry -}} {{- $body := partial "returns/render-returns-md.html" (dict "returns" $entry.returns "cmd" $page.Title) -}} {{- $content = $content | replaceRE `\{\{<\s*return-info(\s+[^>]*)?\s*>\}\}` $body -}} {{- end -}} {{- return $content -}}
